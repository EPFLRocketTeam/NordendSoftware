144 /**
145  * @brief Control thread entry point
146  * @details This thread holds the main state machine of the WildhornAV software. It will be
147  * the main decision point for actions to be taken with respect to real world events.
148  *
149  *
150  * @param arg freertos thread entry point context (unused)
151  *
152  */
153 void control_thread(__attribute__((unused)) void * arg) {
154  static TickType_t last_wake_time;
155  static const TickType_t period = pdMS_TO_TICKS(CONTROL_HEART_BEAT);
156  last_wake_time = xTaskGetTickCount();
157 
158  device_interface_t * hostproc_interface = hostproc_get_interface();
159 
160 
161 
162  for(;;) {
163 
164  led_rgb_set_rgb(0xff, 0, 0);
165  static const uint8_t msg[] = "hello\n\r";
166  device_interface_send(hostproc_interface, msg, sizeof(msg));
167 
168 
169  vTaskDelayUntil( &last_wake_time, period );
170  }
171 }
