59 /**
60  * @brief Compute the CRC for a data packet.
61  * @details Function taken from Maxon Serial V2 specification manual.
62  *
63  * @param p_data_array Pointer to the data array of the packet.
64  * @param length Length of the packet
65  *
66  * @note The p_data_array data must contain two empty bytes [0x00, 0x00] at the end,
67  * where the CRC goes.
68  */
69 uint16_t calc_field_CRC(uint16_t *p_data_array, uint16_t length) {
70  uint16_t shifter, c;
71  uint16_t carry;
72  uint16_t crc = 0;
73  while (length--) {
74  shifter = 0x8000;
75  c = *p_data_array++;
76  do {
77  carry = crc & 0x8000;
78  crc <<= 1;
79  if (c & shifter) crc++;
80  if (carry) crc ^= 0x1021;
81  shifter >>= 1;
82 
83  } while (shifter);
84 
85  }
86  return crc;
87 }

99 /**
100  * @brief Generate an msv2 packet
101  * @details
102  *
103  * @param msv2 Pointer to the msv2 packet creator instance.
104  * @param opcode Opcode to be placed in the packet header.
105  * @param data_len Length of the packet data payload.
106  * @param data Data to be placed in the packet payload.
107  */
108 uint16_t msv2_create_frame(MSV2_INST_t * msv2, uint8_t opcode, uint8_t data_len, uint8_t * data) {
109  uint16_t array_len = data_len+2; //we add 1 for the opcode and len fields and 1 for the crc
110  msv2->tx.data_len = data_len;
111  msv2->tx.opcode = opcode;
112  msv2->tx.data[0] = DLE;
113  msv2->tx.data[1] = STX;
114  msv2->tx.data[2] = opcode;
115  msv2->tx.data[3] = data_len;
116  msv2->tx.crc_data[0] = (data_len<<8) | opcode; //header bytes inverted
117  uint16_t counter=4;
118  for(uint16_t i = 0; i < data_len; i++) {
119  msv2->tx.data[counter++] = data[2*i]; //bytes in data need to be inverted before
120  if(msv2->tx.data[counter-1] == DLE) {
121  msv2->tx.data[counter++] = DLE;
122  }
123  msv2->tx.data[counter++] = data[2*i+1];
124  if(msv2->tx.data[counter-1] == DLE) {
125  msv2->tx.data[counter++] = DLE;
126  }
127  msv2->tx.crc_data[i+1] = (data[2*i+1]<<8) | data[2*i];
128  }
129  msv2->tx.crc_data[array_len-1] = 0x0000;
130  uint16_t crc = calc_field_CRC(msv2->tx.crc_data, array_len);
131  msv2->tx.data[counter++] = crc&0xff; //crc bytes are inverted (LSB first) !!
132  if(msv2->tx.data[counter-1] == DLE) {
133  msv2->tx.data[counter++] = DLE;
134  }
135  msv2->tx.data[counter++] = crc>>8;
136  if(msv2->tx.data[counter-1] == DLE) {
137  msv2->tx.data[counter++] = DLE;
138  }
139  return counter;
140 }

142 /**
143  * @brief Decode an msv2 fragment (one byte at a time)
144  * @details
145  *
146  * @param msv2 Pointer to the msv2 packet creator instance.
147  * @param d Fragment to be decoded.
148  *
149  * @note This function is to be called upon reception of a byte.
150  */
151 MSV2_ERROR_t msv2_decode_fragment(MSV2_INST_t * msv2, uint8_t d) {
152  //if a DLE in data is followed by STX, we start again
153  if (msv2->rx.escape == 1 && d == STX) {
154  msv2->rx.state = WAITING_OPCODE;
155  msv2->rx.escape = 0;
156  return MSV2_PROGRESS;
157  }
158 
159  if (msv2->rx.state == WAITING_DLE && d == DLE) {
160  msv2->rx.state = WAITING_STX;
161  return MSV2_PROGRESS;
162  }
163  //escape in case a DLE is in the data
164  if (d == DLE && msv2->rx.escape == 0) {
165  msv2->rx.escape = 1;
166  return MSV2_PROGRESS;
167  }
168  //if it is doubled, it counts as data
169  if (d == DLE && msv2->rx.escape == 1) {
170  msv2->rx.escape = 0;
171 
172  }
173 
174  if (msv2->rx.state == WAITING_STX && d == STX) {
175  msv2->rx.state = WAITING_OPCODE;
176  return MSV2_PROGRESS;
177  }
178 
179  if (msv2->rx.state == WAITING_OPCODE) {
180  msv2->rx.opcode = d;
181  msv2->rx.state = WAITING_LEN;
182  return MSV2_PROGRESS;
183  }
184 
185  if (msv2->rx.state == WAITING_LEN) {
186  msv2->rx.data_len = d; //legth in words
187  msv2->rx.length = 2*d; //length in bytes
188  msv2->rx.crc_data[0] = (msv2->rx.data_len<<8) | msv2->rx.opcode;
189  msv2->rx.counter = 0;
190  msv2->rx.state = WAITING_DATA;
191  return MSV2_PROGRESS;
192  }
193 
194  if (msv2->rx.state == WAITING_DATA) {
195  msv2->rx.data[msv2->rx.counter] = d;
196  if(msv2->rx.counter & 0x01) { //LSB == '1'
197  msv2->rx.crc_data[msv2->rx.counter/2 + 1] = (msv2->rx.data[msv2->rx.counter - 1]) | d<<8;
198  }
199  msv2->rx.counter += 1;
200  //the length is in WORDS, but we read BYTES
201  if (msv2->rx.counter==msv2->rx.length) {
202  msv2->rx.state = WAITING_CRC1;
203  }
204  return MSV2_PROGRESS;
205  }
206 
207  if (msv2->rx.state == WAITING_CRC1) {
208  msv2->rx.crc = d;
209  msv2->rx.state = WAITING_CRC2;
210  return MSV2_PROGRESS;
211 
212  }
213 
214  if (msv2->rx.state == WAITING_CRC2) {
215  msv2->rx.crc |= d<<8;
216  msv2->rx.state = WAITING_DLE;
217  msv2->rx.crc_data[msv2->rx.data_len+1] = 0;
218  uint16_t crc = calc_field_CRC(msv2->rx.crc_data, msv2->rx.data_len+2);
219  if(msv2->rx.crc == crc) {
220  return MSV2_SUCCESS;
221  } else {
222  return MSV2_WRONG_CRC;
223  }
224  }
225  msv2->rx.state=WAITING_DLE;
226  return MSV2_PROGRESS;
227 }
