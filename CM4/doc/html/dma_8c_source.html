151 void dma_handle_interrupt(dma_stream_dev_t * stream) {
152  //get status flags for the concerned dma stream
153  uint32_t status_flags = 0;
154  if(stream->number < 4) {
155  status_flags = (stream->dma->LISR & (0b11101 << (stream->number*5))) >> (stream->number*5);
156  } else {
157  status_flags = (stream->dma->HISR & (0b11101 << ((stream->number-4)*5))) >> ((stream->number-4)*5);
158  }
159 
160  if(status_flags & DMA_STATUS_TC) { //transfer complete
161  stream->transfer_cplt(stream->user_context);
162  } else if (status_flags & DMA_STATUS_TH) { //transfer half complete
163  stream->transfer_half(stream->user_context);
164  } else if (status_flags & DMA_STATUS_TE) { //transfer error
165  stream->transfer_error(stream->user_context);
166  } else { //unknown situation...
167  //do nothing
168  }
169 }

224 dma_stream_dev_t * dma_scheduler_request_stream(dma_scheduler_dev_t * dma_scheduler) {
225  ENTER_CRITICAL();
226 
227  //check if a stream is available

229  EXIT_CRITICAL();
230  return NULL;
231  }
232 
233  dma_stream_dev_t * stream;
234 
235  for(uint16_t i = 0; i < dma_scheduler->stream_count; i++) {
236  if(dma_scheduler->streams[i]->state == DMA_STREAM_FREE) {
237  stream = dma_scheduler->streams[i];

242  stream->state = DMA_STREAM_BUSY;

263 util_error_t dma_start_stream(dma_stream_dev_t * stream, dma_stream_config_t config) {
264  stream->dma_stream->CR = 0;
265  while (stream->dma_stream->CR && DMA_SxCR_EN); //Wait for transfer to finish
266 
267  if(stream->number < 4) {
268  WRITE_IN_REG(stream->dma->LISR, 0b11101 << (stream->number*5), 0); //clear interrupt flags
269  } else {
270  WRITE_IN_REG(stream->dma->HISR, 0b11101 << ((stream->number-4)*5) , 0); //clear interrupt flags
271  }
272 
273  //set callbacks
274  stream->transfer_cplt = config.transfer_cplt;
275  stream->transfer_half = config.transfer_half;
276  stream->transfer_error = config.transfer_error;
277  stream->user_context = config.user_context;
278  //set addresses
279  stream->dma_stream->PAR = config.p_addr;
280  stream->dma_stream->M0AR = config.m0_addr;
281  stream->dma_stream->M1AR = config.m1_addr;
282 
283  //set transfer size
284  stream->dma_stream->NDTR = config.transfer_size;
285 
286  //set trigger
287  //no synchronization for now!
288 
289  if(config.dmamux_request_number > 127) {
290  return ER_OUT_OF_RANGE;
291  }
292 
293  stream->dmamux_channel->CCR = config.dmamux_request_number;
294 
295 
296  //flow control
297  WRITE_IN_REG(stream->dma_stream->CR, DMA_SxCR_PFCTRL_Msk, config.peripheral_flow_control<<DMA_SxCR_PFCTRL_Pos);
298 
299  //priority
300  WRITE_IN_REG(stream->dma_stream->CR, DMA_SxCR_PL_Msk, config.priority<<DMA_SxCR_PL_Pos);
301 
302  //data direction
303  WRITE_IN_REG(stream->dma_stream->CR, DMA_SxCR_DIR_Msk, config.direction<<DMA_SxCR_DIR_Pos);
304 
305  //enable transfer complete interrupt
306 
307 
308 
309 
310  //start DMA stream
311  stream->dma_stream->CR |= DMA_SxCR_EN;
312 }
