82 /**
83  * @brief Initialize a device interface instance.
84  * @details
85  *
86  * @param interface
87  * @param context
88  * @param deamon
89  * @param send
90  * @param recv
91  * @param handle_data

93 util_error_t device_interface_create( device_interface_t * interface,
94  void * context,
95  device_deamon_t * deamon,
96  util_error_t (*send)(void*, uint8_t*, uint32_t),
97  util_error_t (*recv)(void*, uint8_t*, uint32_t*),
98  util_error_t (*handle_data)(void*, void*))
99 {
100  static int32_t count = 0;
101  interface->context = context;
102  interface->send = send;
103  interface->recv = recv;
104  interface->handle_data = handle_data;
105  interface->id = count++;
106  if(deamon && handle_data) {
107  deamon->interfaces[deamon->interfaces_count] = interface;
108  deamon->interfaces_count++;
109  }
110  return ER_SUCCESS;
111 }

113 /**
114  * @brief Initializea device deamon instance
115  * @details
116  *
117  * @param deamon
118  * @param name
119  * @param prio
120  * @param context
121  * @param data_rdy
122  */
123 util_error_t device_deamon_create( device_deamon_t * deamon,
124  const char * name,
125  uint32_t prio,
126  void * context,
127  util_error_t (*data_rdy)(void*))
128 {
129  static uint32_t counter = 0;
130  deamon->id = counter++;
131  deamon->context = context;
132  deamon->data_rdy = data_rdy;
133  deamon->interfaces_count = 0;
134  if(data_rdy) {
135  deamon->handle = xTaskCreateStatic(device_deamon_thread, name, DEAMON_STACK_SIZE, deamon, prio, deamon->stack, &deamon->buffer);
136  } else {
137  return ER_RESSOURCE_ERROR;
138  }
139  return ER_SUCCESS;
140 }

142 /**
143  * @brief Generic device deamon thread
144  * @details This thread will call the handle data function for an interface,
145  * whenever data is ready for a group of interfaces belonging to
146  * the same deamon.
147  * @param arg FreeRTOS entry point context, used to pass the deamon context to the thread.
148  */
149 void device_deamon_thread(void * arg)
150 {
151  device_deamon_t * deamon = (device_deamon_t * ) arg;
152 
153  for(;;) {
154  if(deamon->data_rdy(deamon->context) == ER_SUCCESS) {
155  //iterate over all interfaces in deamon
156  for(uint16_t i = 0; i < deamon->interfaces_count; i++) {
157  device_interface_t * interface = deamon->interfaces[i];
158  interface->handle_data(interface->context, deamon->context);
159  }
160  }
161  }
162 }
