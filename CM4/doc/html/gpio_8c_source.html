59 void gpio_cfg(GPIO_TypeDef * gpio, uint16_t pins, gpio_config_t cfg) {
60  for (uint16_t position = 0; position < 16; position++) {
61  uint16_t pin = 0b1<<position;
62  if(pins & pin) {
63  WRITE_IN_REG(gpio->ODR, pin, 0); //clear output
64  WRITE_IN_REG(gpio->OTYPER, pin, cfg.drive<<position); //write drive mode
65  WRITE_IN_REG(gpio->OSPEEDR, 0b11<<(position*2), cfg.speed<<(position*2));
66  WRITE_IN_REG(gpio->MODER, 0b11<<(position*2), cfg.mode<<(position*2));
67  if(position < 8) {
68  WRITE_IN_REG(gpio->AFR[0], 0b1111<<(position*4), cfg.alternate<<(position*4));
69  } else {
70  WRITE_IN_REG(gpio->AFR[1], 0b1111<<((position-8)*4), cfg.alternate<<((position-8)*4));
71  }
72 
73  }
74  }
